import { Card, CardGrid } from '@astrojs/starlight/components';

```sh
pip install dysonsphere
```

<CardGrid>
	<Card title="Themed defaults" icon="setting">
		One `ds.theme()` call gives every Altair chart perceptually uniform palettes and
		publication-ready styling.
	</Card>
	<Card title="Composable marks" icon="add-document">
		`mark_strip`, `mark_violin`, statistical annotations, reference lines, and more, all
		returning native Altair objects you compose with `+`.
	</Card>
	<Card title="Round-trippable exports" icon="download">
		`save()` bakes statistics, provenance, and the data into the file; `read()` and `load()`
		pull them (or the whole chart) back out.
	</Card>
	<Card title="Runs in your browser" icon="rocket">
		The [live playground](/playground/) runs dysonsphere via WebAssembly, no install.
	</Card>
</CardGrid>
